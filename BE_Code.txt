program define extrapolate_3step
    syntax varlist

    foreach var of local varlist {
        // ✅ Ensure the variable exists in the dataset
        capture confirm variable `var'
        if _rc != 0 {
            display "⚠️ Variable `var' does not exist. Skipping..."
            continue
        }

        // ✅ **Shorten `_filled` variable name if necessary**
        local shortname = substr("`var'_filled", 1, 31)  // Max 32 characters
        capture confirm variable `shortname'
        if _rc != 0 {
            gen `shortname' = `var'
        }

        // ✅ Iterate for 3-Step Ahead Predictions
        forvalues h = 1/3 {
            // ✅ Generate lag variables dynamically
            capture drop L1_`var' L2_`var' L3_`var'
            gen L1_`var' = L1.`shortname'
            gen L2_`var' = L2.`shortname'
            gen L3_`var' = L3.`shortname'

            // ✅ Check if enough data exists
            count if `shortname' != .
            if r(N) > 10 {
                // ✅ Run an autoregressive regression
                reg `shortname' L1_`var' L2_`var' L3_`var' if `shortname' != .

                // ✅ Predict missing values for this step
                local shortpred = substr("`var'_step`h'", 1, 31)
                capture drop `shortpred'
                predict `shortpred' if `shortname' == .

                // ✅ Fill `_filled` with new predictions
                replace `shortname' = `shortpred' if `shortname' == .
            } 
            else {
                display "⚠️ Skipping `var' (Step `h'): Not enough data for regression."
            }

            // ✅ Drop temporary variables after each step
            capture drop `shortpred' L1_`var' L2_`var' L3_`var'
        }
    }
end

program define extrapolate_missing
    syntax varlist

    foreach var of local varlist {
        // ✅ Ensure the variable exists in the dataset
        capture confirm variable `var'
        if _rc != 0 {
            display "⚠️ Variable `var' does not exist. Skipping..."
            continue
        }

        // ✅ **[CHANGED] Create a shorter new variable name (max 32 characters)**
        local shortname = substr("`var'_filled", 1, 31)  // **Ensures the name does not exceed 32 characters**
        gen `shortname' = `var'

        // ✅ Generate lag variables (limit to 3 lags for stability)
        gen L1_`var' = L1.`var'
        gen L2_`var' = L2.`var'
        gen L3_`var' = L3.`var'

        // ✅ Ensure there are enough observations for regression
        count if `var' != .  // Count non-missing observations
        if r(N) > 10 {  // Require at least 10 observations for a meaningful regression

            // ✅ Run an autoregressive regression only on non-missing values
            reg `var' L1_`var' L2_`var' L3_`var' if `var' != .

            // ✅ **[CHANGED] Predict missing values and store them in a shortened name**
            local shortpred = substr("`var'_ext", 1, 31)  // **Shortened prediction variable**
            predict `shortpred' if `var' == .

            // ✅ **[CHANGED] Fill missing values in the new variable only**
            replace `shortname' = `shortpred' if `shortname' == .  // **Uses the short variable name**
        } 
        else {
            display "⚠️ Skipping `var': Not enough data for regression."
        }

        // ✅ Drop lag variables and temporary extrapolation variable
        drop L1_`var' L2_`var' L3_`var' `shortpred'  // **[CHANGED] Now dropping `shortpred`**
    }
end
